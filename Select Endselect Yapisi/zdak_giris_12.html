REPORT zdak_giris_12.

DATA: connid TYPE sbook-connid,
      fldate          TYPE sbook-fldate,
      gv_ort_tutar    TYPE sbook-forcuram,
      gv_toplam_tutar TYPE sbook-forcuram.
Data gv_sayi type sy-dbcnt.
SELECT connid fldate AVG( forcuram ) SUM( forcuram )
  INTO ( connid, fldate, gv_ort_tutar, gv_toplam_tutar )
  FROM sbook GROUP BY connid fldate.

  gv_sayi = sy-dbcnt.
  write: gv_sayi.
  Write: / 'Uçuþ no:',connid,'Uçuþ Tarihi:', fldate,
           'Tutar ortalamasý:',gv_ort_tutar,'Tutar toplamý:',gv_toplam_tutar.
ENDSELECT.
